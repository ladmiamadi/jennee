import React from 'react'

/**

const SignUpComponentStepThree = ({
    setFormValue,
    formValue,
    HandleSubmit,
    handleChange,
    page,
    HorizontalLabelPositionBelowStepper,
    error,
    setError,
    errorType,
    setErrorType

    return (
